== Verify Install

Now you have _Couchbase Lite for C_ installed, you can verify that your apps build correctly.

Enter the C code in <<ex-clang-starter>> into your editor of choice and build it.

Running it will create a database, add a document, retrieve the document, update the document and then delete the document.

.Sample code
[#ex-clang-starter]
====
[source, c]
----
//
//  main.c
//  clang300testProd


// STILL A WORK IN PROGRESS


#include <stdio.h>
#include "cbl/CBLDatabase.h"
#include "cbl/CBLDocument.h"
#include "cbl/CBLQuery.h"
//#include "cbl/CBLMutableDocument.h"
#include "fleece/Fleece+CoreFoundation.h"

int main(int argc, const char * argv[]) {
    // insert code here...
    printf("Hello from Couchbase Lite for C!\n");

    const char* dbname = "mydb";

    CBLDatabaseConfiguration* dbcfg;

    CBLError* err;
    CBLDatabase* db = CBLDatabase_Open(FLStr(dbname),dbcfg, err);

    CBLDocument* doc = CBLDocument_CreateWithID(FLStr("1000"));

    FLMutableDict* body = FLMutableDict_New();
    FLSlot* slot = FLMutableDict_Set(*body, FLStr("type"));
    FLSlot_SetString(* slot, FLStr("pets"));

    CBLDocument_SetProperties(doc, "type:pet" );

    return 0;
}
